Q3: Applications Processed Per Agent
Collection: applications Viz: Bar chart

[
  {"$match": {"assignedTo": {"$exists": true}, "isDeleted": {"$ne": true}}},
  {"$group": {
    "_id": "$assignedTo",
    "totalProcessed": {"$sum": 1},
    "approved": {"$sum": {"$cond": [{"$in": ["$status", ["APPROVED", "DISBURSED", "CLOSED"]]}, 1, 0]}},
    "rejected": {"$sum": {"$cond": [{"$eq": ["$status", "REJECTED"]}, 1, 0]}}
  }},
  {"$lookup": {"from": "users", "localField": "_id", "foreignField": "_id", "as": "agent"}},
  {"$unwind": {"path": "$agent", "preserveNullAndEmptyArrays": true}},
  {"$project": {
    "agentName": {"$ifNull": ["$agent.fullName", "Unknown"]},
    "totalProcessed": 1,
    "approved": 1,
    "rejected": 1,
    "approvalRate": {"$multiply": [{"$divide": ["$approved", {"$max": ["$totalProcessed", 1]}]}, 100]},
    "_id": 0
  }},
  {"$sort": {"totalProcessed": -1}}
]
